 	loja TINYTEXT NULL,
	valorloja TINYTEXT NULL,
	valorat TINYTEXT NULL,
  	datanota TINYTEXT NULL,

	aparencia LONGTEXT NULL,
	problema LONGTEXT NULL,
	bateria TINYTEXT NULL,
	adulteracao TINYTEXT NULL,

)
